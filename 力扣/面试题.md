## 面试题 01.05. 一次编辑

字符串有三种编辑操作:插入一个字符、删除一个字符或者替换一个字符。 给定两个字符串，编写一个函数判定它们是否只需要一次(或者零次)编辑。

```js
/**
 * @param {string} first
 * @param {string} second
 * @return {boolean}
 */
var oneEditAway = function(first, second) {
    let m = first.length;
    let n = second.length;
    if (n - m ===1) {
        return oneInsert(first, second);
    }else if(m - n ===1){
        return oneInsert(second, first);
    }else if (m === n){
        let diff=false;
        for (let i = 0; i < m; i++){
            if (first[i] !== second[i]) {
                if (!diff) {
                    diff = true;
                }else {
                    return false;
                }
            }
        }
        return true;
    }else{
        return false;
    }
};

var oneInsert = function(shorter, longer){
    let len1=shorter.length, len2=longer.length;
    let index1 = 0, index2 = 0;
    while (index1 < len1 && index2 < len2){
        //比较对应下标处的字符是否相同，如果字符相同则将两个字符串的下标同时加 1
        if (shorter[index1] === longer[index2]) {
            index1++;
        }
        //如果字符不同则只将 longer 的下标加 1
        index2++;
        if (index2 - index1 > 1) {
            return false;
        }
    }
    return true;
}

let first = "pale", second = "ple";
let result = oneEditAway(first, second);
console.log(result);
```

## 面试题 04.06. 后继者

设计一个算法，找出二叉搜索树中指定节点的“下一个”节点（也即中序后继）。

如果指定节点没有对应的“下一个”节点，则返回null。

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @param {TreeNode} p
 * @return {TreeNode}
 */
var inorderSuccessor = function(root, p) {
    let arr = [];
    let prev = null, curr = root;
    while (arr.length || root){
        while (curr){
            arr.push(curr);
            curr = curr.left;
        }
        curr = arr.pop();
        if (prev === p) {
            return curr;
        }
        prev = curr;
        curr = curr.right;
    }
    return null;
};
```

利用二叉搜索树的性质

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @param {TreeNode} p
 * @return {TreeNode}
 */
var inorderSuccessor = function(root, p) {
    let successor = null;
    if (p.right) {
        //如果节点 p 的右子树不为空，则节点 p 的后继节点在其右子树中
        successor = p.right;
        //在其右子树中定位到最左边的节点，即为节点 p 的后继节点。
        while (successor.left) {
            successor = successor.left;
        }
        return successor;
    }
    let node = root;
    //如果节点 p 的右子树为空，则需要从根节点开始遍历寻找节点 p 的祖先节点。
    while (node)  {
        if (node.val > p.val) {
            //如果 node 的节点值大于 pp 的节点值，则 p 的后继节点可能是node 或者在 node 的左子树中
            successor = node;
            node = node.left;
        }else {
            //如果 node 的节点值小于或等于 p 的节点值，则 p 的后继节点可能在 node 的右子树中，因此将 node 移动到其右子节点继续遍历。
            node = node.right;
        }
    }
    return successor;
};
```
